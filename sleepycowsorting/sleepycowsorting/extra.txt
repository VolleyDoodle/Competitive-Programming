

#include <iostream>
#include <fstream>
using namespace std;

int N;
int A[100000];

int main()
{
	ifstream fin("sleepy.in");
	ofstream fout("sleepy.out");
	fin >> N;
	for (int i = 0; i < N; i++)
		fin >> A[i];
	int ans = N - 1;
	for (int i = N - 2; i >= 0; i--)
	{
		if (A[i] < A[i + 1])
			ans = i;
		else
			break;
	}
	fout << ans << '\n';
	return 0;
}
